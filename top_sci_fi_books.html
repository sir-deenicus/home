Most popular Scifi

Sunday, November 03, 2013

The most favored Scifi author according to ~535 guardian readers is Iain M. Banks and the most favored book is Dune (wtf, is fantasy not sci-fi ;).

I was recently looking for some scifi books to read and stumbled upon this link. As is not useful, each individual recommendation is not as meaningful as knowing what people favored most. So I spent a few minutes writing a script to get a better idea of what was buried in the data.

This wasn't completely straightforward because people have many different ways of spelling author and book names. For example, people have trouble spelling Ursula K le Guin's name - with 6 variations spotted:

["ursula k le guin"; "ursula k leguin"; "ursula k mcguinn"; "ursula laguinn"; "ursula le guin"; "ursula leguin"]

So had to write a heuristic to get around that for books and names. Here are a couple it spotted I was impressed by:

["heinlein"; "heinlin?"; "ra heinlein"; "robert a. heinlein"; "robert a.heinlen"; "robert heinlein"]

["evgenij zamjatin"; "yevgeny zamyatin"]

["asimov"; "isaac asimov"; "isaac assimove"]

The algorithm isn't perfect, I think it missed 2 or 3 equivalent names and maybe 2 books. There were about 6 spots where one of author or book name were missing, one where author and book were swapped and one interesting mention of Moby Dick by Herman Melville. But in aggregate, certainly not enough to affect stats (and I am quit happy with the results given how little time I spent on the code).

In order, the top 5 authors are: Iain M Banks, Philip K Dick, Frank Herbert, Isaac Asimov, Robert A. Heinlein.

The top 10 books are: Dune, Player Of Games, The Star My Destination, Foundation, Left Hand Of Darkness, Lord Of Light, Ender's Game, Ubik, The Man In The High Castle, The Disposessed.

From my poor intersection with this list I can see that my taste in sci-fi is very particular (hard) - of the top 20, I've only read from Asimov. Although, I have read other genre work of Zelanzy [Amber series] and Le Guin [Earthsea trilogy (best magic system other than Deathgate Cycle)] but those of course are fantasy. In terms of sci-fi I'd much rather read Vinge than Gibson, Egan than Heinlein and Stross than Stephenson.

But I think I'll pursue Iain's books, Diamond Age and Lord of Light - I liked Zelanzy other stuff, and the Culture books & The Diamond age look quite interesting.

The top books no doubt are biased from the Matthew Effect (rich get richer) so some other time I'll post an analysis of one of Stross threads where he asks for the most profound books (this one was raw text so had a trickier time - had to use Active learning to teach author and book names).

let txt = @"..scifilist.txt" |> IO.File.ReadAllLines

let auths = txt |> Array.map (fun s -> s |> splitstr [|"\t"|] 
                                         |> flip Array.get 1 
                                         |> trim
                                         |> tolower) |> set |> Set.remove ""

let books = txt |> Array.map (fun s -> s |> splitstr [|"\t"|] 
                                         |> flip Array.get 0 
                                         |> trimWith superflouschars 
                                         |> tolower) |> set |> Set.remove "" 

let rec buildIt limlev limdic buildSet = function
  | [] -> buildSet
  | currentWord :: supplySet ->
      let take, leave = supplySet |> List.partition (fun s ->  
          //printfn "%A %A" currentWord s
          let ws1 = splitstr [|" "|] currentWord |> Array.filter (String.length >> ((<) 1))
          let ws2 = splitstr [|" "|] s |> Array.filter (String.length >> ((<) 1)) 

          let dice,lev = 
           match (ws1.Length, ws2.Length)  with
            | l1, l2 when l1 = l2 -> Array.map2 stringSimilarityDice ws1 ws2 |> Array.average, 
                                     Array.map2 LevenshteinDistance ws1 ws2 |> Array.average
            | 1, _  -> stringSimilarityDice ws1.[0] (ws2.[ws2.Length - 1]), 
                                  LevenshteinDistance ws1.[0] (ws2.[ws2.Length - 1])
            | _, 1 -> stringSimilarityDice ws2.[0] (ws1.[ws1.Length - 1]), 
                                  LevenshteinDistance ws2.[0] (ws1.[ws1.Length - 1])
            | 3, 2 ->  Array.map2 stringSimilarityDice [|ws1.[0];ws1.[2]|] ws2 |> Array.average,
                       Array.map2 LevenshteinDistance [|ws1.[0];ws1.[2]|] ws2 |> Array.average
            | 2, 3 ->  Array.map2 stringSimilarityDice [|ws2.[0];ws2.[2]|] ws1 |> Array.average,
                       Array.map2 LevenshteinDistance [|ws2.[0];ws2.[2]|] ws1 |> Array.average
            | _ ->  stringSimilarityDice currentWord s, LevenshteinDistance currentWord s

          dice > limdic || lev < limlev)

      buildIt limlev limdic ((currentWord::take)::buildSet) leave

let rec buildItBooks limlev limdic buildSet = function
  | [] -> buildSet
  | currentWord :: supplySet ->
      let take, leave = supplySet |> List.partition (fun s ->   
          let dice = stringSimilarityDice currentWord s//, LevenshteinDistance currentWord s 
          dice > limdic )//|| lev < limlev)
      buildItBooks limlev limdic ((currentWord::take)::buildSet) leave      

let l = buildIt 1. 0.69 [] (Set.toList auths)   
let lookup = l |> List.map (Set.ofList) 

let l2 = buildItBooks 2. 0.8 [] (Set.toList books) 
//let l2 = buildIt 1. 0.81 [] (Set.toList books)  
let lookupBk = l2 |> List.map (Set.ofList) 

let res =
    txt |> Array.map (fun s -> let dat = s |> splitstr [|"\t"|]
                               dat.[0] |> trim, dat.[1] |> trim)
        |> Seq.groupBy (fun (bookname,bookauthor) ->  
                        if bookauthor = "" then "Author Missing"
                        else
                          lookup |> List.find (Set.contains (tolower bookauthor)) |> Seq.head)
        |> Seq.toArray
        |> Array.map (fun (author, booklist) ->
               let bklist = booklist |> Seq.countBy (fst >> tolower >> (trimWith superflouschars) >>
                                                    (fun b -> if b = "" then "" else lookupBk |> List.find (Set.contains (tolower b)) |> Seq.head))
                                     |> Seq.toList  
               booklist |> Seq.head |> snd, bklist, bklist |> Seq.sumBy snd)
        |> Array.sortByDescending third

res.[..4] |> Array.map fst3 |> joinToStringWith ", "

let bres = 
 res |> Array.toList |> List.collect snd3 |> Seq.groupBy fst
     |> Seq.map (fun (book , bks) -> book, bks |> Seq.sumBy snd)
     |> Seq.sortByDescending snd
     |> Seq.toArray
bres.Length

bres.[..9] |> Array.map (fst >> capitilizeFirst) |> joinToStringWith ", "

let bkhtml = bres |> Array.map (fun (b,c) -> sprintf "<li>%s, %d</li>" (b |> capitilizeFirst) c) |> joinToStringWith newLine

let bhtml = 
    res |> Array.mapi (fun i (auth, bks, total) -> 
              let bkstr = bks |> List.sortByDescending snd |> List.map (fun (b,c) -> sprintf "<li>%s: %d</li>"  (b |> capitilizeFirst) c) |> joinToString   
              sprintf "<h3>%d. %s - %d</h3><ul>%s</ul>" (i+1) auth total bkstr) |> joinToStringWith newLine

let dir = @"..documents"                                           
IO.File.WriteAllText (dir + "top scifi.htm", bhtml)
IO.File.WriteAllText (dir + "topbooks.html","<ul>" + bkhtml + "</ul>")
